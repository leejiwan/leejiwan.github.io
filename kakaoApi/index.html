var kakaoAuth = ""; //kakao restApikey
function loadDoc() {
  $.ajax({
    method: "GET",
    url: "https://dapi.kakao.com/v2/search/image?",
    data: { query: "가스공사" },
    headers: { Authorization: "KakaoAK " + kakaoAuth },
  })
    .done(function (msg) {
      alert("Data Saved: " + msg);
    })
    .fail(function (error) {});
}
